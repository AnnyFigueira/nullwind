# createForm

`createForm` is a function that creates a form object. It takes a schema and returns a form object.

## Usage

```jsx
import Nullstack from "nullstack";

import { Button, Input, percentFilled } from "nullwind";
import { object, string } from "yup";

const validationSchema = object({
  firstName: string().required("First Name is required"),
});

export default class Preview extends Nullstack {
  initiate() {
    this.form = createForm({ validationSchema });
  }

  async send() {
    await this.form.validate();

    if (this.form.isValid) {
      console.log("Form is valid!");
    }
  }

  render() {
    return (
      <form onsubmit={this.send}>
        <Input
          label="Last Name"
          bind={this.form.data.firstName}
          error={this.form.errors.firstName}
        />
        <Button type="submit">Submit</Button>
      </form>
    );
  }
}
```
